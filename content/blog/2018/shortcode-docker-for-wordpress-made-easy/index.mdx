Just copy this into a `docker-compose.yml` in your project root, and then run `docker-compose up -d` üöÄ

```yml
version: '2'
services:
  my-wpdb:
    image: mariadb
    ports:
      - '8081:3306'
    environment:
      MYSQL_ROOT_PASSWORD: root
  my-wp:
    image: wordpress
    volumes:
      - ./:/var/www/html
    ports:
      - '8080:80'
    links:
      - my-wpdb:mysql
    environment:
      WORDPRESS_DB_PASSWORD: root
```
